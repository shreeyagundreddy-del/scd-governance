CREATE OR REPLACE PROCEDURE rollback_version(
    p_customer_id VARCHAR,
    p_target_version INT,
    p_reason TEXT
)
LANGUAGE plpgsql
AS $$
BEGIN
    -- Close current version
    UPDATE dim_customer
    SET is_current_flag = FALSE,
        effective_to = CURRENT_DATE - 1
    WHERE customer_id = p_customer_id
      AND is_current_flag = TRUE;

    -- Re-activate target version
    UPDATE dim_customer
    SET is_current_flag = TRUE,
        effective_to = '9999-12-31',
        change_reason = p_reason
    WHERE customer_id = p_customer_id
      AND version_number = p_target_version;
END;
$$;
